L = 3.6
E_beam = 290 # MeV/u
mc2 = 940 # MeV
c = 3.0e8 # [m/s]


def dzdtheta(theta):
  h = L*np.sin(theta0)

  # return h/(np.tan(theta)**2*np.cos(theta)**2)
  return h/np.sin(theta)**2

def gamma(E): # E [MeV/u]
  return 1 + E/mc2
def beta(E):
  return np.sqrt(gamma(E)**2-1)/gamma(E)

def P(theta, Ep):
  # phits

  return np.exp(-3/((120/180)*np.pi)*theta)*(theta > np.pi/6) + np.exp(-3/((120/180)*np.pi)*np.pi/6)*(theta <= np.pi/6) #  Ep = 50

import bisect

Ep = 50 # MeV
bins_theta = np.linspace(theta0, np.pi*3/4, num=100000)
bins_t = np.linspace(0, 50, num=100)
# print(bins_t)
delta_theta = bins_theta[1] - bins_theta[0]
delta_t = bins_t[1] - bins_t[0]


f_t = np.zeros(bins_t.shape)

for theta in bins_theta:
  # print(bins_t)
  # print(t(theta,Ep))
  i_t = bisect.bisect_left(bins_t, t(theta, Ep))
  # print(i_t)
  if i_t == len(f_t):
    continue
  f_t[i_t] += f_theta(theta, Ep)*delta_theta/delta_t
